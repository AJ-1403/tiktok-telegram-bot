# ------------------------------------------------------------
# Multi‑stage build for a tiny production image
# ------------------------------------------------------------
FROM python:3.11-slim AS builder

# Install git (needed only for build) and build-essential for yt‑dlp
RUN apt-get update && apt-get install -y --no-install-recommends \
    git \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /app

# Copy only the dependency file first (helps Docker cache)
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# ------------------------------------------------------------
# Runtime image
# ------------------------------------------------------------
FROM python:3.11-slim

WORKDIR /app

# Copy only the necessary runtime files
COPY --from=builder /usr/local/lib/python3.11/site-packages /usr/local/lib/python3.11/site-packages
COPY bot.py .
COPY README.md .

# Create a non‑root user for security
RUN useradd -m botuser && chown -R botuser:botuser /app
USER botuser

# The bot token must be supplied at runtime
ENV TELEGRAM_BOT_TOKEN=YOUR_TOKEN_HERE

CMD ["python", "bot.py"]
